def binary_search(lists, key):
    '''判空、仅有1个、排序、low< high、没有查找到'''二分查找法，又叫折半查找，应用于等概分布的有序序列，利用了判定树，平均查找长度 ASL = (log2 n+1) - 1
    if len(lists) == 0:
        return -1
    if len(lists) == 1:
        if key == lists[0]:
            return 0
        else:
            return -1
    new = sorted(lists)
    low, high = 0, len(new) - 1
    while low < high:
        mid = (high - low) / 2
        if new[mid] > key:
            high = mid - 1
        elif new[mid] < key:
            low = mid + 1
        elif new[mid] == key:
            return mid
    return -1
